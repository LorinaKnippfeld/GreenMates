exports.addPlant = (plant_id, user_id, common_name, scientific_name, image_url) => {
    return db.query(
        `INSERT INTO plant_garden
            (plant_id, user_id, common_name, scientific_name, image_url) 
        VALUES 
               ($1, $2, $3, $4, $5) 
        RETURNING 
            *
            `,
        [plant_id, user_id, common_name, scientific_name, image_url]
    );
};

// Delete friend request

exports.deletePlant = (id) => {
    return db.query(
        "DELETE FROM plant_garden WHERE id=$1;",
        [id]
    );
};

exports.getPlants = (user_id) => {
     return db.query(
        `SELECT * FROM plant_garden WHERE user_id=$1 
        ORDER BY created_at DESC`,
        [user_id]
    );
};
